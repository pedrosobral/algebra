\newcommand*\BuildChpNum[2]{%
	\begin{tabular}[t]{@{}c@{}}
	\makebox[0pt][c]{\strut} \\[.5ex]
	\colorbox{black}{%
	\rule[-10em]{0pt}{0pt}%
	\rule{.7ex}{0pt}\color{white}\scalebox{1.5}{#2}\strut
	\rule{.5ex}{0pt}
}%
	\end{tabular}}
